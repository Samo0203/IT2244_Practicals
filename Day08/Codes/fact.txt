Q2)Factorial
// Output: Factorial of 5 is: 120

code:-

// Initialize fact to 1
fact=1

// Loop from i=1 to i<=5
for((i=1; i<=5; i++))
do
    // Multiply current fact with i and store it back in fact
    fact=$(($fact * $i))
done

// Output the result
echo "Factorial of 5 is: "$fact
